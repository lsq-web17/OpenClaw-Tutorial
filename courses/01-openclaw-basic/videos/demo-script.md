#### 演示4：创建Hello World技能（90秒）
```bash
# 创建技能目录
cd ~/.openclaw/workspace
mkdir -p skills/hello-world
cd skills/hello-world

# 创建技能文件
cat > skill.py << 'EOF'
#!/usr/bin/env python3
from datetime import datetime

class HelloSkill:
    def handle_message(self, msg):
        if 'hello' in msg.lower():
            return "👋 你好！我是你的OpenClaw助手！"
        elif 'time' in msg.lower():
            return f"🕐 当前时间：{datetime.now().strftime('%H:%M:%S')}"
        return "🤔 试试说'hello'或'time'？"

skill = HelloSkill()
print(skill.handle_message("hello"))
print(skill.handle_message("time"))
EOF

# 运行技能
python3 skill.py
```

